dy = [-1, 0, 1, 0]
dx = [0, 1, 0, -1]
def dfs(y, x):
    global path
    global max_path
    global use_chance
    global K
    for d in range(4):
        next_y = y + dy[d]
        next_x = x + dx[d]
        if not (0 <= next_y < N and 0 <= next_x < N):
            continue
        if visited[next_y][next_x] != 0:
            continue
        if Map[y][x] <= Map[next_y][next_x]:
            if use_chance:
                continue
            if Map[y][x] > Map[next_y][next_x] - K:
                tmp = Map[next_y][next_x]
                Map[next_y][next_x] = Map[y][x] -1
                use_chance = True
                visited[next_y][next_x] = 1
                path += 1
                if path > max_path:
                    max_path = path
                dfs(next_y, next_x)
                use_chance = False
                Map[next_y][next_x] = tmp
                path -= 1
                visited[next_y][next_x] = 0
                continue
            else:
                continue
        visited[next_y][next_x] = 1
        path += 1
        if path > max_path:
            max_path = path
        dfs(next_y, next_x)
        path -= 1
        visited[next_y][next_x] = 0
tc = int(input())
for test_ in range(1, tc + 1):
    N, K = map(int, input().split())
    Map = [list(map(int, input().split())) for _ in range(N)]
    visited = [[0]*N for _ in range(N)]
    use_chance = False
    max_h = 0
    path = 0
    max_path = 0
    for y in range(N):
        for x in range(N):
            if Map[y][x] > max_h:
                max_h = Map[y][x]
    for y in range(N):
        for x in range(N):
            if Map[y][x] == max_h:
                visited[y][x] = 1
                dfs(y, x)
                visited[y][x] = 0
    print('#{} {}'.format(test_, max_path+1))
