tc = int(input())
for test_ in range(1, tc + 1):
    n = int(input())
    lis = [list(map(int, input().split())) for _ in range(n)]
    ret = [[] for _ in range(n)]
    for i in range(n):
        s90 = ''
        s180 = ''
        s270 = ''
        for ii in range(n):
            s90 += str(lis[n-1-ii][i])
            s180 += str(lis[n-1-i][n-1-ii])
            s270 += str(lis[ii][n-1-i])
        ret[i].append(s90)
        ret[i].append(s180)
        ret[i].append(s270)
    print('#{}'.format(test_))
    for i in range(n):
        print(' '.join(ret[i]))
