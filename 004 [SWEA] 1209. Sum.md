    row_sum = [0 for i in range(202)]
    for i in range(100):
        for ii in range(100):

            row_sum[100+ii] += lis[i][ii]
            if i == ii:
                row_sum[200] += lis[i][ii]
            if 100 - i - 1 == ii:
                row_sum[201] += lis[i][ii]
    print('#{} {}'.format(test_, max(row_sum)))
