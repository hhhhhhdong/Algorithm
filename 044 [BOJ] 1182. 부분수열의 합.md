```python
N, S = map(int, input().split())
lis = list(map(int, input().split()))
ans = 0
for i in range(1, 1 << N):
    tmp = 0
    for ii in range(N):
        if i & 1 << ii:
            tmp += lis[ii]
    if tmp == S:
        ans += 1
    tmp = 0
print(ans)
```
