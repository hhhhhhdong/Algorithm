def right(x, y, lis):
    while lis[y+1][x] == 0:
        x -= 1
    return x, y
def left(x, y, lis):
    while lis[y+1][x] == 0:
        x += 1
    return x, y

for test_ in range(1, 11):
    n = int(input())
    lis = [list(map(int, input().split())) for _ in range(100)]
    for i in range(100):
        if lis[0][i] == 0:
            continue
        x = i
        y = 0
        while not y == 99:
            if x - 1 > 0:
                if lis[y][x-1] == 1:
                    x -= 1
                    x, y = right(x, y, lis)
                    y += 1
                    continue
            if x + 1 <= 99:
                if lis[y][x+1] == 1:
                    x += 1
                    x, y = left(x, y, lis)  # x=1 y=2
                    y += 1
                    continue
            y += 1
        if lis[y][x] == 2:
            print('#{} {}'.format(test_, i))
            break
