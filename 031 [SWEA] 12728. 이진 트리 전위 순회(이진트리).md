def dfs(n):
    if n == 0:
        return
    ret.append(n)
    dfs(left[n])
    dfs(right[n])
tc = int(input())
for test_ in range(1, tc + 1):
    v = int(input())
    left = [0 for _ in range(v+1)]
    right = [0 for _ in range(v+1)]
    ret = []
    for i in range(v-1):
        p, c = map(int, input().split())
        if left[p] == 0:
            left[p] = c
        else:
            right[p] = c
    dfs(1)
    print('#{} '.format(test_), end='')
    print(*ret)
