    stack = []
    result = ''
    for i in s:

            result += i

            if i == ')':

                stack.pop()
            else:
                while stack and icp[i] <= isp[stack[-1]]:

                stack.append(i)
    while stack:
        result += stack.pop()
    for i in result:
        if ord('0') < ord(i) <= ord('9'):
            stack.append(i)
        else:
            a = int(stack.pop())
            b = int(stack.pop())
            if i == '-':
                stack.append(b - a)
            elif i == '+':
                stack.append(b + a)

                stack.append(b * a)
            elif i == '/':
                stack.append(b / a)
