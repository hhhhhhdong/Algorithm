def pool(m, cost):
    global ans
    if m >= 12:
        if ans > cost:
            ans = cost
        return
    pool(m+1, cost+D*month[m])
    pool(m+1, cost+M1)
    pool(m+3, cost+M3)
tc = int(input())
for test_ in range(1, tc + 1):
    D, M1, M3, Y = map(int, input().split())
    month = list(map(int, input().split()))
    ans = Y
    pool(0, 0)
    print('#{} {}'.format(test_, ans))
