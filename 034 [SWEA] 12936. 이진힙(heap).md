tc = int(input())
for test_ in range(1, tc + 1):
    N = int(input())
    lis = list(map(int, input().split()))
    tree = [0]
    ret = 0
    for i in range(N):
        tree.append(lis[i])
        idx = len(tree) - 1
        while idx // 2 > 0 and tree[idx] < tree[idx//2]:
            tree[idx], tree[idx//2] = tree[idx//2], tree[idx]
            idx //= 2
    while N // 2 > 0:
        ret += tree[N//2]
        N //= 2
    print('#{} {}'.format(test_, ret))
