from collections import deque

dy = [-1, 0, 1, 0]
dx = [0, 1, 0, -1]


def findStart():
    for y in range(n):
        for x in range(n):

def BFS(sy, sx):
    q.append([sy, sx])
    visited[sy][sx] = 1
    while q:
        y, x = q.popleft()
        for i in range(4):
            ny = y + dy[i]
            nx = x + dx[i]
            if 0 > ny or 0 > nx or n <= ny or n <= nx: continue
            if visited[ny][nx] != 0: continue
            if Map[ny][nx] == 1: continue
            if Map[ny][nx] == 3:
                return visited[y][x]
            q.append([ny, nx])
            visited[ny][nx] = visited[y][x] + 1


tc = int(input())

    q = deque()
    sy, sx = findStart()
