def pascal(n, c=1, before=[1]):
    temp_lis = [0] * c
    for i in range(c):
        if i == 0 or i == c-1:
            temp_lis[i] = 1
        else:
            temp_lis[i] = before[i-1] + before[i]
    print(*temp_lis)
    if c == n:
        return
    return pascal(n, c+1, temp_lis)


tc = int(input())
for test_ in range(1, tc + 1):
    print('#{}'.format(test_))
    n = int(input())
    pascal(n)
