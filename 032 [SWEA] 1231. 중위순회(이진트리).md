def search(n):
    if n:
        search(left[n])
        print(word[n], end='')
        search(right[n])

for test_ in range(1, 11):
    n = int(input())
    word = [''] * (n+1)
    left = [0] * (n+1)
    right = [0] * (n+1)
    for i in range(1, n+1):
        s = input().split()
        word[i] = s[1]
        if len(s) == 4:
            left[i] = int(s[2])
            right[i] = int(s[3])
        if len(s) == 3:
            left[i] = int(s[2])
    root = 1
    while left[root] == 0 and right[root] == 0:
        root += 1
    print('#{} '.format(test_), end=' ')
    search(root)
    print()
