```python
def paper(w):
    global cnt
    if memo[w] != 0:
        return memo[w]
    cnt += 1
    if w == 1:
        memo[w] = 1
        return 1
    elif w == 2:
        memo[w] = 3
        return 3
    memo[w] = paper(w-1) + 2*paper(w-2)
    return memo[w]
tc = int(input())
for test_ in range(1, tc + 1):
    n = int(input()) // 10
    memo = [0] * (n+1)
    paper(n)
    print('#{} {}'.format(test_, memo[n]))
```
