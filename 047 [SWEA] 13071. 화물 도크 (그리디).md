def sort_key(v):
    return (v[1], v[0])
for test_ in range(1, int(input()) + 1):
    N = int(input())
    lis = [list(map(int, input().split())) for _ in range(N)]
    lis.sort(key=sort_key)
    cnt = 0
    end = 0
    for i in range(N):
        if end <= lis[i][0]:
            end = lis[i][1]
            cnt += 1
    print('#{} {}'.format(test_, cnt))
