tc = int(input())
for test_ in range(1, tc + 1):
    s = input()
    start = 0
    ans = 0
    i = 0
    while i < len(s):
        if i < len(s) - 1 and s[i] == '(' and s[i+1] == ')':
            ans += start
            i += 1
        elif s[i] == '(':
            start += 1
        else:
            ans += 1
            start -= 1
        i += 1
    print('#{}'.format(test_), ans)
