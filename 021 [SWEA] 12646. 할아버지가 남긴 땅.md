def robot(x, y):
    sx = x
    sy = y
    dx = [1, 0, -1, 0]
    dy = [0, 1, 0, -1]
    now_sum = 0
    for d in range(4):
        while Map[y+dy[d]][x+dx[d]]:
            y += dy[d]
            x += dx[d]
            now_sum += Map[y][x]
            if d == 2 and sx == x:
                break
            if d == 3 and sy == y:
                break

    return now_sum


tc = int(input())
for test_ in range(1, tc + 1):
    h, w = map(int, input().split())
    Map = [list(map(int, input().split())) + [0] for _ in range(h)]
    Map.append([0] * (w+1))
    max_sum = 0
    for y in range(h):
        for x in range(w):
            if Map[y][x]:
                temp = robot(x, y)
                if temp > max_sum:
                    max_sum = temp
