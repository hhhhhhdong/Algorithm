def sub_sum(level, total, idx):
    global cnt
    if level == n:
        if total == k:
            cnt += 1
        return
    if total >= k:
        return
    for i in range(idx, 20):
        sub_sum(level+1, total+i+1, i+1)

tc = int(input())
for test_ in range(1, tc+1):
    n, k = map(int, input().split())
    cnt = 0
    sub_sum(0, 0, 0)
    print('#{} {}'.format(test_, cnt))
